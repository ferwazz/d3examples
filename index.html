Data Driven Documents

D3 allows document elements to be generated from data. Hence the name Data Driven Documents (D3).

The example below shows the distribution of people by age in the United States.

const canvas  = d3.select(".canva");

height = 600
width = 800

const svg = canvas.append("svg")
                .attr("width", width)
                .attr("height", height);

const margin = {top: 20, bottom: 70, left: 70, right: 70};
const graphWidth = 700 - margin.left - margin.right;
const graphHeight = 700 - margin.top - margin.bottom;

var formatComma = d3.format(",");

    
const mainCanvas = svg.append("g")
                        .attr("width", graphWidth / 2)
                        .attr("height", graphHeight / 2)
                        .attr("transform", `translate(${margin.left + 300}, 
                            ${margin.top + 290})`);

                            
mainCanvas.append("text")
          .attr("class", "daca-text")
          .attr("dy", ".85cm")
          .style("opacity", 0.7)
          .transition()
                .duration(1000)
                .style("opacity", (d,i) => i+0.7)
          .text("US Population per age")
          .attr("text-anchor", "middle")


let n = 17
colorbyValue = {}
 
 for (let i = 0; i < n; ++i) {
   colorbyValue[data[i].name] = d3.rgb(d3.interpolateSpectral(i / (n - 1))).hex();
 }


var arcs = d3.pie()(data.map(function(d) { return d.value; }));

const radius = height / 2;

const arcpath = d3.arc()
                .innerRadius(radius * 0.6)
                .outerRadius(radius - 10)

const paths =  mainCanvas.selectAll("path")
                         .data(arcs)


paths.enter()
    .append("path")
    .attr("class", "arc")
    .attr("stroke", "white")
    .attr("fill", function(d, i){return colorbyValue[data[i].name]})
    .attr("d", arcpath)
